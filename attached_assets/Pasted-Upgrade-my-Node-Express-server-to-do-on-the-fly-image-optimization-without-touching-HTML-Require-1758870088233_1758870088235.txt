import express from "express";
import fs from "fs";
import path from "path";
import sharp from "sharp";

const app = express();
const __dirname = path.dirname(new URL(import.meta.url).pathname);

// ---- Image optimizer (place BEFORE static) ----
app.use(async (req, res, next) => {
  try {
    // Only handle .jpg/.jpeg/.png
    if (!/\.(jpe?g|png)$/i.test(req.path)) return next();

    const originalAbs = path.join(__dirname, "public", req.path);
    if (!fs.existsSync(originalAbs)) return next(); // let static 404 it

    // Decide output format via Accept header
    const accept = req.headers["accept"] || "";
    const fmt = accept.includes("image/avif") ? "avif" : "webp";
    const quality = fmt === "avif" ? 55 : 70;
    const maxWidth = 1200;

    // Build cache key using file mtime
    const stat = fs.statSync(originalAbs);
    const cacheKey = [
      originalAbs,
      Math.floor(stat.mtimeMs),
      fmt,
      quality,
      `w${maxWidth}`
    ].join("|");

    const cacheDir = path.join(__dirname, ".img-cache");
    await fs.promises.mkdir(cacheDir, { recursive: true });
    const cacheName =
      Buffer.from(cacheKey).toString("hex") + "." + fmt;
    const outAbs = path.join(cacheDir, cacheName);

    if (!fs.existsSync(outAbs)) {
      let pipeline = sharp(originalAbs).rotate().resize({
        width: maxWidth,
        withoutEnlargement: true
      });
      pipeline =
        fmt === "avif"
          ? pipeline.avif({ quality })
          : pipeline.webp({ quality });
      await pipeline.toFile(outAbs);
    }

    res.setHeader("Content-Type", `image/${fmt}`);
    res.setHeader("Cache-Control", "public, max-age=31536000, immutable");
    res.setHeader("Vary", "Accept");

    fs.createReadStream(outAbs).pipe(res);
  } catch (e) {
    // On error, fall back to normal static handling
    next();
  }
});

// ---- Your existing static handler (keep as-is) ----
app.use(express.static(path.join(__dirname, "public")));
