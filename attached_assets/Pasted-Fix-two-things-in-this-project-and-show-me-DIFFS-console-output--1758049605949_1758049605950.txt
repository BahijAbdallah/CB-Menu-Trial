(function(){
  let scroller = document.getElementById('categoryStrip');
  if(!scroller){ console.warn('categoryStrip not found'); return; }

  // Unblock any parent overflow that prevents horizontal scroll
  let p = scroller.parentElement;
  while(p){
    const cs = getComputedStyle(p);
    if(cs.overflowX === 'hidden') p.style.overflowX = 'visible';
    p = p.parentElement;

  scroller.style.overflowX = 'auto';

  function updateArrows(){

    leftBtn.disabled = atStart;

WITH n AS (
  SELECT id, name, LOWER(TRIM(name)) AS k,
         CASE WHEN image IS NOT NULL AND image<>'' THEN 1 ELSE 0 END AS has_img,
         COALESCE(createdAt, id) AS ord
  FROM items
),
w AS (
  SELECT k,
         COALESCE(MAX(CASE WHEN has_img=1 THEN id END),
                  MIN(id)) AS keep_id
  FROM n GROUP BY k
)
-- Repoint links
UPDATE item_category SET item_id = (SELECT keep_id FROM w WHERE w.k = LOWER(TRIM((SELECT name FROM items WHERE items.id = item_category.item_id)))) 
WHERE EXISTS (SELECT 1 FROM w WHERE w.keep_id IS NOT NULL);
-- Delete dups
DELETE FROM items WHERE id NOT IN (SELECT keep_id FROM w) AND LOWER(TRIM(name)) IN (SELECT k FROM w);

- **Postgres** (adjust table/columns if needed):
WITH n AS (
  SELECT id, name, LOWER(TRIM(name)) AS k,
         (CASE WHEN image IS NOT NULL AND image<>'' THEN 1 ELSE 0 END) AS has_img,
         COALESCE(created_at, id) AS ord
  FROM items
),
w AS (
  SELECT DISTINCT ON (k)
         k, COALESCE( max(id) FILTER (WHERE has_img=1), min(id) ) AS keep_id
  FROM n GROUP BY k
)
UPDATE item_category ic
SET item_id = w.keep_id
FROM items i
JOIN w ON LOWER(TRIM(i.name)) = w.k
WHERE ic.item_id = i.id AND i.id <> w.keep_id;

DELETE FROM items i
USING w
WHERE LOWER(TRIM(i.name)) = w.k AND i.id <> w.keep_id;
